<?xml version="1.0" encoding="UTF-8"?>

<!--
    ARQUIVO: persistence.xml

    RESPONSABILIDADE:
    Configurar as unidades de persistência (persistence-unit) do JPA/Hibernate.

    OBSERVAÇÃO IMPORTANTE:
    Cada persistence-unit representa uma configuração de acesso a um banco de dados.

    NESTE EXEMPLO:
    - SQLitePU     -> configuração para SQLite
    - MySQLPU      -> configuração para MySQL
    - PostgresPU   -> configuração para PostgreSQL
    - SqlServerPU  -> configuração para SQL Server

    USO:
    No código Java, escolha qual unidade será usada, por exemplo:
    Persistence.createEntityManagerFactory("SQLitePU");
-->

<persistence xmlns="https://jakarta.ee/xml/ns/persistence"
             xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
             xsi:schemaLocation="https://jakarta.ee/xml/ns/persistence
             https://jakarta.ee/xml/ns/persistence/persistence_3_0.xsd"
             version="3.0">

    <!-- ========================================================= -->
    <!-- SQLITE                                                    -->
    <!-- ========================================================= -->
    <persistence-unit name="SQLitePU" transaction-type="RESOURCE_LOCAL">

        <!--
            Provider JPA utilizado.
            Aqui estamos usando o Hibernate como implementação do JPA.
        -->
        <provider>org.hibernate.jpa.HibernatePersistenceProvider</provider>

        <!--
            Classe de entidade que será mapeada pelo Hibernate.
            Esta classe normalmente usa @Entity.
        -->
        <class>br.unisales.database.table.Usuario</class>

        <properties>

            <!--
                Driver JDBC do SQLite.
                É o driver responsável por abrir a conexão com o banco.
            -->
            <property name="jakarta.persistence.jdbc.driver" value="org.sqlite.JDBC"/>

            <!--
                URL de conexão com o SQLite.

                IMPORTANTE:
                Este caminho é relativo ao diretório de execução da aplicação.
                Se desejar, pode usar caminho absoluto também.

                Exemplo relativo:
                ./biblioteca/dados/biblioteca_db.db

                Exemplo absoluto:
                D:/projetos/java/biblioteca/dados/biblioteca_db.db
            -->
            <property name="jakarta.persistence.jdbc.url" value="jdbc:sqlite:./biblioteca/dados/biblioteca_db.db"/>

            <!--
                Dialect do Hibernate para SQLite.
                O dialect informa ao Hibernate como gerar SQL compatível com o banco.
            -->
            <property name="hibernate.dialect" value="org.hibernate.community.dialect.SQLiteDialect"/>

            <!--
                hbm2ddl.auto controla a criação/atualização automática das tabelas.

                Valores comuns:
                - none        -> não faz nada
                - create      -> apaga e cria tudo novamente
                - update      -> cria/atualiza tabelas sem apagar os dados
                - create-drop -> cria ao iniciar e apaga ao encerrar

                Para aula, geralmente "update" é uma boa opção.
            -->
            <property name="hibernate.hbm2ddl.auto" value="update"/>

            <!--
                Exibe o SQL gerado no console.
                Muito útil para fins didáticos.
            -->
            <property name="hibernate.show_sql" value="true"/>

            <!--
                Formata o SQL exibido no console.
                Facilita a leitura.
            -->
            <property name="hibernate.format_sql" value="true"/>

        </properties>
    </persistence-unit>

    <!-- ========================================================= -->
    <!-- MYSQL                                                     -->
    <!-- ========================================================= -->
    <persistence-unit name="MySQLPU" transaction-type="RESOURCE_LOCAL">

        <provider>org.hibernate.jpa.HibernatePersistenceProvider</provider>

        <class>br.unisales.database.table.Usuario</class>

        <properties>

            <!-- Driver JDBC do MySQL -->
            <property name="jakarta.persistence.jdbc.driver" value="com.mysql.cj.jdbc.Driver"/>

            <!--
                URL de conexão do MySQL.

                localhost  -> servidor MySQL local
                3306       -> porta padrão do MySQL
                biblioteca -> nome do banco de dados

                Ajuste o nome do banco conforme necessário.
            -->
            <property name="jakarta.persistence.jdbc.url" value="jdbc:mysql://localhost:3306/biblioteca?useSSL=false&amp;serverTimezone=America/Sao_Paulo"/>

            <!-- Usuário do banco MySQL -->
            <property name="jakarta.persistence.jdbc.user" value="root"/>

            <!-- Senha do banco MySQL -->
            <property name="jakarta.persistence.jdbc.password" value="123456"/>

            <!-- Dialect do Hibernate para MySQL -->
            <property name="hibernate.dialect" value="org.hibernate.dialect.MySQLDialect"/>

            <!-- Criação/atualização automática das tabelas -->
            <property name="hibernate.hbm2ddl.auto" value="update"/>

            <!-- Mostrar SQL no console -->
            <property name="hibernate.show_sql" value="true"/>

            <!-- Formatar SQL -->
            <property name="hibernate.format_sql" value="true"/>

        </properties>
    </persistence-unit>
